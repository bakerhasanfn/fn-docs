    Filter Framework Implementation Guide¶

    This guide walks through the process of creating a new data view using the Filter Framework. You'll learn how to create templates, plugins, and model properties to build a complete feature.

    Implementation Process Overview¶

    graph TD
    A[Define Requirements] --> B[Create Template]
    B --> C[Create Plugin]
    C --> D[Create Model Properties]
    D --> E[Create Factory]
    E --> F[Register Route]
    F --> G[Create Controller]
    G --> H[Create View]
    H --> I[Test]

    Prerequisites¶

    Before starting, make sure you understand: - The Filter Framework request lifecycle - The Component Model architecture - Basic PHP object-oriented programming

    Implementation Steps¶

    1. Define Your Requirements¶

    Start by clearly defining: - What data will be displayed - How it should be filtered - What sorting options are needed - Which views are required (list, model, map, etc.)

    2. Create a Template Class¶

    Templates define the high-level structure of your data view:

    1. Create a new PHP file in modules/Filter/Template/ directory
    2. Define a class that extends AbstractTemplate
    3. Implement the init() method
    <?php
namespace Filter\Template;

class MyCustomData extends AbstractTemplate
{
    protected $defaultView = 'model';
    protected $plugins = array();
    protected $mapType = 'c'; // Custom identifier for map view

    public function init()
    {
        parent::init();

        // Register plugins
        $this->plugins = array(\Filter\Plugin\MyCustomData::class => false);

        // Set default columns for list view
        $this->setDefaultColumns(array(
            'id',
            'name',
            'category',
            'created_date',
            'status',
        ));

        // Add help text
        $this
            ->setHelpTitle('My Custom Data')
            ->setHelpCategory('My Category')
            ->setHelpDescription('View and filter custom data.');
    }
}

    3. Create a Plugin Class¶

    Plugins handle the data querying and transformation:

    1. Create a new PHP file in modules/Filter/Plugin/ directory
    2. Define a class that extends AbstractPlugin
    3. Implement the init() and filter() methods
    <?php
namespace Filter\Plugin;

use Filter\Component;

class MyCustomData extends AbstractPlugin
{
    // Define optional components
    protected $optionalComponents = array(
        'details',
        'history'
    );

    // Define model properties to use
    protected $models = array();

    public function init()
    {
        // Create model properties
        $this->models = array(
            new MyCustomDataModelProperties\Base($this),
            new MyCustomDataModelProperties\Details($this),
            new MyCustomDataModelProperties\Status($this),
        );

        // Set up model and filters
        $this->setupModel();
        $this->setupFilters();

        // Add segment filter
        $this->add((new Component\Filterable('Segment', 'My Custom Data'))
            ->setOrder(1)
            ->setAlwaysActive(true)
        );

        // Add status filter
        $this->add((new Component\Filterable('General', 'Status'))
            ->setCallable('post', function($enabled = false, $value = null) {
                if ($enabled && $value !== null) {
                    $this->addWhere('c.status_id = :status_id');
                    $this->addParam('status_id', $value);
                }
            })
        );
    }

    protected function setupModel()
    {
        foreach ($this->models as $model) {
            $model->modelComponents();
        }

        return $this;
    }

    protected function setupFilters()
    {
        foreach ($this->models as $model) {
            $model->filterableComponents();
        }

        return $this;
    }

    public function filter()
    {
        // Build and execute the query to populate the result pool
        $sql = sprintf(
            "SELECT
                c.id AS id
            FROM
                my_custom_data c
            %s
            WHERE
                c.company_id = :company_id
            %s
            %s
            %s",
            $this->join(),
            $this->where(),
            $this->having(),
            $this->order()
        );

        $stmt = $this->get('dbRo')->prepare($sql);
        $stmt->bindValue('company_id', $this->get('companyId'), \PDO::PARAM_INT);
        $this->bindParams($stmt);
        $stmt->execute();

        $ids = array();
        while ($row = $stmt->fetch(\PDO::FETCH_ASSOC)) {
            $ids[] = $row['id'];
        }

        $this->replacePool($ids);
    }

    public function gather()
    {
        $results = array();

        if (!$this->getPool()) {
            return $results;
        }

        $sql = sprintf(
            "SELECT
                c.id AS id,
                c.name AS name,
                c.category_id AS category_id,
                c.created_date AS created_date,
                c.status_id AS status_id,
                s.name AS status_name
            FROM
                my_custom_data c
                LEFT JOIN status s ON s.id = c.status_id
            %s
            WHERE
                c.id IN (%s)
            %s
            %s
            %s
            LIMIT %d OFFSET %d",
            $this->join(),
            $this->escapeIds($this->getPool()),
            $this->where(),
            $this->having(),
            $this->order(),
            $this->getTemplate()->getResultsPerPage(),
            $this->getTemplate()->getPaginationOffset()
        );

        $stmt = $this->get('dbRo')->prepare($sql);
        $this->bindParams($stmt);
        $stmt->execute();

        while ($row = $stmt->fetch(\PDO::FETCH_ASSOC)) {
            $results[$row['id']] = $row;
        }

        return $results;
    }

    public function postProcess($id, array $row)
    {
        // Process each model
        foreach ($this->models as $model) {
            $row = $model->postProcess($id, $row);
        }

        return $row;
    }
}

    4. Create Model Property Classes¶

    Model properties organize related components:

    1. Create a new directory modules/Filter/Plugin/MyCustomDataModelProperties
    2. Create model property classes for each domain concept
    <?php
namespace Filter\Plugin\MyCustomDataModelProperties;

use Filter\Component;

class Base extends AbstractMyCustomDataModelProperty
{
    public function modelComponents()
    {
        // ID component
        $this->add((new Component\Displayable('Json', 'id'))
            ->setAlias('id')
            ->setCallable('json', $this->jsonInt('id'))
        );

        // Name component
        $this->add((new Component\Displayable('Json', 'name'))
            ->setAlias('name')
            ->setCallable('query', function() {
                $this->addSelect('c.name AS name');
            })
            ->setCallable('json', $this->jsonTrim('name'))
        );

        // Created date component
        $this->add((new Component\Displayable('Json', 'created_date'))
            ->setAlias('created_date')
            ->setCallable('query', function() {
                $this->addSelect('c.created_date AS created_date');
            })
            ->setCallable('json', function($row) {
                return array(
                    'created_date' => array(
                        'date' => $row['created_date'],
                        'formatted' => date('Y-m-d H:i:s', strtotime($row['created_date']))
                    )
                );
            })
        );
    }

    public function filterableComponents()
    {
        // Name filter
        $this->add((new Component\Filterable('Search', 'Name'))
            ->setCallable('post', function($enabled = false, $value = null) {
                if ($enabled && $value !== null) {
                    $this->addWhere('c.name LIKE :name');
                    $this->addParam('name', "%$value%");
                }
            })
        );

        // Date range filter
        $this->add((new Component\Filterable('Search', 'Date Range'))
            ->setCallable('post', function($enabled = false, $value = null) {
                if ($enabled && is_array($value)) {
                    if (!empty($value['from'])) {
                        $this->addWhere('c.created_date >= :date_from');
                        $this->addParam('date_from', $value['from'] . ' 00:00:00');
                    }
                    if (!empty($value['to'])) {
                        $this->addWhere('c.created_date <= :date_to');
                        $this->addParam('date_to', $value['to'] . ' 23:59:59');
                    }
                }
            })
        );
    }
}

{

}

    7. Create a Controller¶

    Create a controller to handle the requests:

    <?php
namespace Filter\Controller;

use FieldNation\Framework\Controller;

class MyCustomData extends Controller
{
    public function list()
    {
        $this->isRequired('VIEW_MY_CUSTOM_DATA');

        $template = $this->bootstrapAngularFilterTemplate(
            'Filter\Template\MyCustomData',
            array(
                'route' => '/my-custom-data/list',
            )
        );

        return $this->render(
            'my-custom-data/list.html.twig',
            array(
                'template' => $template,
            ),
            'layout/app.html.twig'
        );
    }
}

    8. Create Twig Templates¶

    Create the necessary Twig templates for rendering:

    {# View/my-custom-data/list.html.twig #}
{% extends layout %}

{% block title %}My Custom Data{% endblock %}

{% block content %}
<div class="my-custom-data">
    <div class="row">
        <div class="col-md-12">
            <h1>My Custom Data</h1>

            <div ng-app="FilterApp">
                <div ng-controller="FilterController">
                    <div fn-filter="{{ template.route }}" ng-model="filterData"></div>

                    <div class="results-container">
                        <div ng-repeat="item in filterData.results" class="result-item">
                            <h3>{{ "{{ item.name }}" }}</h3>
                            <div class="details">
                                <span class="status">{{ "{{ item.status.name }}" }}</span>
                                <span class="date">{{ "{{ item.created_date.formatted }}" }}</span>
                            </div>
                        </div>
                    </div>
                </div>
            </div>
        </div>
    </div>
</div>
{% endblock %}

    Integration with Other Services¶

    To integrate with other services:

    // In your Plugin class:
public function gather()
{
    $results = parent::gather();

    // Enhance with data from other services
    $service = $this->get('myService');
    foreach ($results as &$row) {
        $row['additional_data'] = $service->getData($row['id']);
    }

    return $results;
}

    Common Pitfalls¶

    1. Memory Usage: Be careful with large result sets
    2. Performance: Optimize SQL queries and joins
    3. Reusing Components: Don't duplicate component definitions
    4. SQL Injection: Always use parameterized queries
    5. UI Consistency: Follow the existing UI patterns
    6. Session Management: Be aware of session state
