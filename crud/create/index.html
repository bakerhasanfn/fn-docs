    Creating Records with Filter Framework¶

    The Filter Framework provides a structured approach to implement Create operations for your data entities. This document outlines the steps and components needed to implement creation functionality for your custom entities.

    Overview¶

    Creating new records in the Filter Framework typically involves:

    1. Defining data model properties
    2. Creating a dedicated service for record creation
    3. Setting up a controller to handle creation requests
    4. Implementing proper validation and error handling
    5. Integrating with the database layer

    File Structure¶

    Before implementation, understand the directory structure where your files should be placed:

    fieldnation/
├── library/
│   └── FieldNation/          # Core framework code
├── modules/
│   └── YourModule/           # Your module code
│       ├── Controller/       # Controllers
│       ├── Factory/
│       │   └── Service/      # Service factories
│       ├── Model/            # Entity models
│       ├── ModelProperties/  # Properties for models
│       ├── Service/          # Service implementations
│       ├── Api/
│       │   └── v2/           # API controllers
│       │
│       ├── routes.php        # Route definitions
│       └── v2_api.php        # API route definitions
│
└── view/
    └── YourModule/           # Twig templates

    Implementation Process¶

    1. Define Your Data Model¶

    Before implementing creation functionality, define the data structure using model properties:

    <?php
namespace YourModule\ModelProperties;

use Filter\Component;

class YourEntityProperties
{
    public function defineProperties()
    {
        // Define properties for creation
        return [
            'name' => [
                'type' => 'string',
                'required' => true,
                'validation' => 'notEmpty'
            ],
            'description' => [
                'type' => 'string',
                'required' => false
            ],
            'status_id' => [
                'type' => 'integer',
                'required' => true
            ],
            'created_date' => [
                'type' => 'dateTime',
                'defaultValue' => 'CURRENT_TIMESTAMP'
            ]
        ];
    }
}

    For model properties in the Filter Framework, you typically create them in the modules/YourModule/ModelProperties directory. These properties define the data structure and validation rules for your entities.

    2. Create a Service for Record Creation¶

    Create a dedicated service that handles the creation of new records. The services should be placed in the modules/YourModule/Service directory:

    <?php
namespace YourModule\Service;

use FieldNation\Framework\Service;
use FieldNation\Framework\Service\Traits;

class CreateEntity extends Service
{
    use Traits\User; // For user validation

    // Properties to be set by caller
    protected $name;
    protected $description;
    protected $statusId;

    /**
     * Execute the creation process
     * 
     * @return array Result with success/failure and new entity ID
     */
    public function execute()
    {
        // Validate the current user
        $this->validateUser();

        // Validate input data
        $this->validateInputs();

        // Begin transaction
        $db = $this->get('db');
        $db->beginTransaction();

        try {
            // Insert the record
            $db->insert('your_entity_table', [
                'name' => $this->name,
                'description' => $this->description,
                'status_id' => $this->statusId,
                'created_by' => $this->get('userId'),
                'company_id' => $this->get('companyId')
            ]);

            // Get the newly created entity ID
            $entityId = $db->lastInsertId();

            // Commit the transaction
            $db->commit();

            // Return success response
            return [
                'success' => true,
                'entity_id' => $entityId
            ];
        } catch (\Exception $e) {
            // Rollback on error
            $db->rollback();

            // Return error response
            return [
                'success' => false,
                'message' => $e->getMessage()
            ];
        }
    }

    /**
     * Validate the input data
     */
    protected function validateInputs()
    {
        if (empty($this->name)) {
            throw new \InvalidArgumentException('Name is required');
        }

        if (empty($this->statusId)) {
            throw new \InvalidArgumentException('Status ID is required');
        }
    }

    // Setters for properties
    public function setName($name)
    {
        $this->name = $name;
        return $this;
    }

    public function setDescription($description)
    {
        $this->description = $description;
        return $this;
    }

    public function setStatusId($statusId)
    {
        $this->statusId = (int)$statusId;
        return $this;
    }
}

    3. Create a Factory for Your Service¶

    Register a factory to create your service. The factory should be placed in the modules/YourModule/Factory/Service directory:

    <?php
namespace YourModule\Factory\Service;

use FieldNation\Framework\ServiceManager;
use YourModule\Service\CreateEntity;

class CreateEntityFactory implements ServiceManager\FactoryInterface
{
    public function create(ServiceManager\Locator $app)
    {
        $service = new CreateEntity();
        $service->setDb($app->get('db'));
        $service->setUserId($app->get('userId'));
        $service->setCompanyId($app->get('companyId'));

        return $service;
    }
}

    4. Create a Controller for Handling Create Requests¶

    Define a controller to handle the creation requests. Create this file in the modules/YourModule/Controller directory:

    <?php
namespace YourModule\Controller;

use FieldNation\Framework\Controller;

class EntityController extends Controller
{
    /**
     * Create a new entity
     */
    public function create()
    {
        // Check permissions
        $this->isRequired('CREATE_YOUR_ENTITY');

        // Get request parameters
        $params = $this->getContext()->getRequest();

        // Create entity
        $result = $this->get(\YourModule\Service\CreateEntity::class)
            ->setName($params['name'])
            ->setDescription($params['description'] ?? '')
            ->setStatusId($params['status_id'])
            ->execute();

        if ($result['success']) {
            // Redirect to the newly created entity or return success
            return $this->redirect('/your-module/entity/' . $result['entity_id']);
        } else {
            // Handle error case
            return $this->render(
                'your-module/create.html.twig',
                [
                    'error' => $result['message'],
                    'formData' => $params
                ]
            );
        }
    }

    /**
     * Show the entity creation form
     */
    public function showCreateForm()
    {
        // Check permissions
        $this->isRequired('CREATE_YOUR_ENTITY');

        // Render the creation form
        return $this->render(
            'your-module/create.html.twig',
            [
                'statuses' => $this->get('statusService')->getAll()
            ]
        );
    }
}

    5. Register Routes for Create Operations¶

    Add routes to handle the creation form and submission. Create or modify your module's routes.php file in the modules/YourModule directory:

    <?php
// In your module's routes.php
use FieldNation\Framework\Route;

return [
    new Route(
        'your-module/entity/create',
        [\YourModule\Controller\EntityController::class, 'showCreateForm'],
        [],
        ['methods' => ['GET']]
    ),
    new Route(
        'your-module/entity/create',
        [\YourModule\Controller\EntityController::class, 'create'],
        [],
        ['methods' => ['POST']]
    )
];

    Integrating with the Filter Framework¶

    Using the REST API for Creation¶

    For API-based creation, implement a REST controller in the modules/YourModule/Api/v2 directory:

    <?php
namespace YourModule\Api\v2;

use FieldNation\Framework\RestController;

class EntityController extends RestController
{
    /**
     * Create a new entity via API
     * 
     * @return \FieldNation\Framework\Response\Json
     */
    public function create()
    {
        // Check authorization
        $this->isGranted();

        // Get JSON request body
        $params = $this->getJsonParams();

        // Validate required fields
        if (empty($params['name']) || empty($params['status_id'])) {
            return $this->error('Name and status_id are required fields');
        }

        // Create entity
        $result = $this->get(\YourModule\Service\CreateEntity::class)
            ->setName($params['name'])
            ->setDescription($params['description'] ?? '')
            ->setStatusId($params['status_id'])
            ->execute();

        if ($result['success']) {
            // Return the newly created entity
            return $this->json([
                'success' => true,
                'entity_id' => $result['entity_id']
            ]);
        } else {
            // Return error
            return $this->error($result['message']);
        }
    }
}

    Register the API route in your module's v2_api.php file:

    <?php
// In your module's v2_api.php
use FieldNation\Framework\Route;

return [
    new Route(
        'api/v2/your-module/entities',
        [\YourModule\Api\v2\EntityController::class, 'create'],
        [],
        ['methods' => ['POST']]
    )
];

    For more comprehensive API implementations, you might want to utilize the RouteBuilder helper:

    <?php
// In your module's v2_api.php
use FieldNation\Framework\Route;
require_once(__DIR__ . '/../Filter/Service/v2/RouteBuilder.php');

return array_merge(
    \Filter\Service\v2\RouteBuilder::build('your-module/entities', \YourModule\Api\v2\EntityController::class),
    // Additional routes...
);

    Integration with Angular Frontend¶

    If using Angular for the frontend, here's how you can integrate with your create API:

    // In your Angular controller
app.controller('CreateEntityController', ['$scope', '$http', '$location', 
    function($scope, $http, $location) {
        $scope.entity = {};
        $scope.statuses = [];
        $scope.loading = false;
        $scope.error = null;

        // Load statuses for dropdown
        $http.get('/api/v2/statuses').then(function(response) {
            $scope.statuses = response.data;
        });

        // Create entity function
        $scope.createEntity = function() {
            $scope.loading = true;
            $scope.error = null;

            $http.post('/api/v2/your-module/entities', $scope.entity)
                .then(function(response) {
                    if (response.data.success) {
                        // Redirect to the entity list or detail page
                        $location.path('/your-module/entities/' + response.data.entity_id);
                    } else {
                        $scope.error = response.data.message || 'Failed to create entity';
                    }
                    $scope.loading = false;
                })
                .catch(function(error) {
                    $scope.error = error.data && error.data.message 
                        ? error.data.message 
                        : 'An error occurred while creating the entity';
                    $scope.loading = false;
                });
        };
    }
]);

    Error Handling and Validation¶

    For robust creation functionality, implement proper validation:

    1. Input Validation: Validate all input fields before attempting to create a record
    2. Use type validation (shouldBeString(), shouldBeNumber(), etc.)
    3. Check for required fields

    4. Validate format (email, phone, etc.)

    5. Permission Checks: Ensure the user has the proper permissions

    6. Use isRequired() in controllers
    7. Use isGranted() in REST controllers

    8. Implement row-level permissions if needed

    9. Database Transactions: Use transactions to ensure data consistency

    10. Begin transaction before operations
    11. Commit only after all operations succeed

    12. Roll back on any failure

    13. Error Messages: Provide meaningful error messages for users

    14. Be specific about what failed
    15. Avoid exposing system details

    16. Consider internationalization

    17. Logging: Log errors for debugging purposes

    18. Include context for troubleshooting
    19. Log user actions for audit

    Best Practices¶

    1. Separate Concerns: Keep creation logic separate from filtering logic
    2. Use dedicated services for create operations

    3. Don't mix CRUD operations in a single service

    4. Use Services: Encapsulate creation logic in dedicated services

    5. Follow the Service pattern with proper factories

    6. Implement the execute() method for consistency

    7. Parameterized Queries: Use parameterized queries to prevent SQL injection

    8. Never concatenate user input into SQL

    9. Use the framework's database abstraction methods

    10. ACL Integration: Integrate with the ACL system for permissions

    11. Define clear permission requirements in your controllers

    12. Implement row-level permissions when necessary

    13. Event Logging: Log creation events for audit purposes

    14. Record who created what and when

    15. Include enough context for accountability

    16. Validation Rules: Define clear validation rules for each field

    17. Document validation constraints for API consumers
    18. Provide helpful validation error messages

    Example: Implementing a Custom Field Creator¶

    Here's a complete example for creating custom fields using the Filter Framework:

    <?php
namespace CustomFields\Service;

use FieldNation\Framework\Service;
use FieldNation\Framework\Service\Traits;

class CreateCustomField extends Service
{
    use Traits\User;

    // Properties for custom field creation
    protected $name;
    protected $type;
    protected $required = false;
    protected $unique = false;
    protected $tip;
    protected $category;
    protected $forProvider = false;
    protected $visibleToClients = false;
    protected $shownInHeader = false;
    protected $includeInAlerts = false;
    protected $options = [];

    /**
     * Execute the creation of a custom field
     * 
     * @return array Result with success/failure and new custom field ID
     */
    public function execute()
    {
        // Validate user permissions
        $this->validateUser();

        // Validate required input
        if (empty($this->name)) {
            throw new \InvalidArgumentException('Custom field name is required');
        }

        if (empty($this->type)) {
            throw new \InvalidArgumentException('Custom field type is required');
        }

        // Begin transaction
        $db = $this->get('db');
        $db->beginTransaction();

        try {
            // Insert the custom field
            $db->insert('workorder_custom_labels', [
                'company_id' => $this->get('companyId'),
                'label' => $this->name,
                'tip' => $this->tip,
                'data_field_type' => $this->type,
                'required' => $this->required ? 1 : 0,
                'unique' => $this->unique ? 1 : 0,
                'fortech' => $this->forProvider ? 1 : 0,
                'hide_from_tech' => $this->forProvider ? 0 : 1,
                'in_alerts' => $this->includeInAlerts ? 1 : 0,
                'show_in_wo_details_header' => $this->shownInHeader ? 1 : 0, 
                'visible_to_clients' => $this->visibleToClients ? 1 : 0,
                'created_by' => $this->get('userId'),
                'created_date' => date('Y-m-d H:i:s')
            ]);

            // Get the newly created custom field ID
            $customFieldId = $db->lastInsertId();

            // Add options for predefined (dropdown) type fields
            if ($this->type === 'predefined' && !empty($this->options)) {
                foreach ($this->options as $index => $option) {
                    $db->insert('custom_label_value_choices', [
                        'custom_label_id' => $customFieldId,
                        'value' => $option,
                        'value_order' => $index
                    ]);
                }
            }

            // Add configuration entry to make the field visible in UI
            $db->insert('workorder_configuration', [
                'company_id' => $this->get('companyId'),
                'client_company_id' => 0,
                'workorder_group_id' => 0,
                'workorder_id' => 0,
                'option_category' => 1, // Custom field category
                'option_name' => $customFieldId,
                'option_value' => 1
            ]);

            // Commit the transaction
            $db->commit();

            return [
                'success' => true,
                'custom_field_id' => $customFieldId
            ];
        } catch (\Exception $e) {
            // Rollback on error
            $db->rollback();

            return [
                'success' => false,
                'message' => $e->getMessage()
            ];
        }
    }

    // Setters with fluent interface
    public function setName($name)
    {
        $this->name = $name;
        return $this;
    }

    public function setType($type)
    {
        $this->type = $type;
        return $this;
    }

    public function setRequired($required)
    {
        $this->required = (bool)$required;
        return $this;
    }

    public function setUnique($unique)
    {
        $this->unique = (bool)$unique;
        return $this;
    }

    public function setTip($tip)
    {
        $this->tip = $tip;
        return $this;
    }

    public function setCategory($category)
    {
        $this->category = $category;
        return $this;
    }

    public function setForProvider($forProvider)
    {
        $this->forProvider = (bool)$forProvider;
        return $this;
    }

    public function setVisibleToClients($visible)
    {
        $this->visibleToClients = (bool)$visible;
        return $this;
    }

    public function setShownInHeader($shown)
    {
        $this->shownInHeader = (bool)$shown;
        return $this;
    }

    public function setIncludeInAlerts($include)
    {
        $this->includeInAlerts = (bool)$include;
        return $this;
    }

    public function setOptions(array $options)
    {
        $this->options = $options;
        return $this;
    }
}

} else {
    // Handle error
